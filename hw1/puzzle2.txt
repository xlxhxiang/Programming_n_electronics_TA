//define all functions here
function moveDown(int n){

	int steps = 0;

	while (steps < n){

		down;
		steps++;

	}

}

function moveRight(int n){

	int steps = 0;

	while (steps < n){

		right;

	}

}

function moveUp(int n){

	int steps = 0;

	while (steps < n){

		up;

	}
}

//main code runs start here
moveDown(6);
moveRight(3);
moveUp(2);
moveRight(2);
moveDown(3);
